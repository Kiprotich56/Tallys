### 3. Appointment Conflict Detection Missing

**Problem:** Staff can be double-booked, same time slot given to multiple customers

**Implementation (appointments.ts):**

```typescript
import { eq, and, lte, gte } from "drizzle-orm";

async function checkConflict(
  staffId: number, 
  date: string, 
  timeSlot: string, 
  durationMinutes: number
) {
  const [service] = await db
    .select({ durationMinutes: servicesTable.durationMinutes })
    .from(servicesTable)
    .where(eq(servicesTable.id, serviceId));

  const slotMinutes = timeSlot.split(":").reduce((h, m) => h * 60 + parseInt(m));
  const endMinutes = slotMinutes + (service?.durationMinutes ?? 30);

  // Get all appointments for this staff on this date
  const conflicts = await db
    .select()
    .from(appointmentsTable)
    .where(
      and(
        eq(appointmentsTable.staffId, staffId),
        eq(appointmentsTable.date, date),
        eq(appointmentsTable.status, "confirmed") // Don't count cancelled
      )
    );

  for (const conflict of conflicts) {
    const existingMinutes = conflict.timeSlot.split(":").reduce((h, m) => h * 60 + parseInt(m));
    const existingEnd = existingMinutes + (service?.durationMinutes ?? 30);
    
    if (slotMinutes < existingEnd && endMinutes > existingMinutes) {
      return true; // Conflict detected
    }
  }
  return false;
}

// In POST /appointments
router.post("/appointments", requireAuth, async (req, res) => {
  const parsed = CreateAppointmentBody.safeParse(req.body);
  if (!parsed.success) { res.status(400).json({ error: parsed.error.message }); return; }

  // Check for conflicts
  const hasConflict = await checkConflict(
    parsed.data.staffId,
    parsed.data.date,
    parsed.data.timeSlot,
    30 // TODO: fetch from service
  );

  if (hasConflict) {
    res.status(409).json({ error: "This time slot is unavailable for selected staff" });
    return;
  }

  // Continue with booking...
});
```

```typescript
import { pgTable, text, serial, timestamp, integer, real } from "drizzle-orm/pg-core";

export const paymentsTable = pgTable("payments", {
  id: serial("id").primaryKey(),
  appointmentId: integer("appointment_id"), // optional, allows partial payments
  customerId: integer("customer_id").notNull(),
  amountKes: integer("amount_kes").notNull(),

  status: text("status").notNull(), // "pending", "success", "failed"
  mpesaCheckoutId: text("mpesa_checkout_id"),
  transactionRef: text("transaction_ref"),
  notes: text("notes"),
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
  completedAt: timestamp("completed_at", { withTimezone: true }),
});
```

```typescript
    .where(eq(paymentsTable.mpesaCheckoutId, checkoutRequestId));

      .set({ status: "success", completedAt: new Date() })

        .set({ status: "confirmed" })
        .where(eq(appointmentsTable.id, payment.appointmentId));
    }
  } else {
    await db.update(paymentsTable)
      .set({ status: "failed" })
      .where(eq(paymentsTable.id, payment.id));
  }
});
```

```typescript
router.post("/auth/register", async (req, res) => {
```

**Create lib/queue/reminder-queue.ts:**
```typescript
import { Queue, Worker } from "bullmq";
import { sendWhatsAppMessage } from "../whatsapp";

const reminderQueue = new Queue("appointment-reminders", {
  connection: { host: process.env.REDIS_HOST || "localhost" },
});

export async function scheduleReminder(appointmentId: number, appointmentDate: string) {
  const [dateStr, timeStr] = [appointmentDate, "10:00"];
  const reminderTime = new Date(`${dateStr}T${timeStr}`);
  reminderTime.setHours(reminderTime.getHours() - 24); // 24 hours before

  await reminderQueue.add(
    "send-reminder",
    { appointmentId },
    { scheduledAt: reminderTime.getTime() }
  );
}

export const reminderWorker = new Worker("appointment-reminders", async (job) => {
  const { appointmentId } = job.data;
  
  const [appt] = await db.select().from(appointmentsTable)
    .where(eq(appointmentsTable.id, appointmentId));
  
  const [customer] = await db.select().from(customersTable)
    .where(eq(customersTable.id, appt.customerId));

  if (customer.phone) {
    await sendWhatsAppMessage(
      customer.phone,
      `Hi ${customer.name}, reminder: your appointment at Tally's tomorrow at ${appt.timeSlot}`
    );
  }
}, { connection: { host: process.env.REDIS_HOST || "localhost" } });
```

### 8. Pagination on All List Endpoints

**Pattern to implement everywhere:**

```typescript
router.get("/appointments", async (req, res) => {
  const { limit = 20, offset = 0 } = req.query;
  const pageLimit = Math.min(parseInt(limit as string) || 20, 100);
  const pageOffset = Math.max(parseInt(offset as string) || 0, 0);

  const [rows, [{ count }]] = await Promise.all([
    db.select().from(appointmentsTable)
      .limit(pageLimit)
      .offset(pageOffset),
    db.select({ count: sql`count(*)` }).from(appointmentsTable),
  ]);

  res.json({
    data: rows,
    pagination: {
      total: count,
      limit: pageLimit,
      offset: pageOffset,
      hasMore: pageOffset + pageLimit < count,
    },
  });
});
```

**Add to OpenAPI spec:**
```yaml
parameters:
  - name: limit
    in: query
    schema: { type: integer, default: 20, maximum: 100 }
  - name: offset
    in: query
    schema: { type: integer, default: 0 }
```

### 10. Audit Logs

**Schema (lib/db/src/schema/audit-logs.ts):**
```typescript
export const auditLogsTable = pgTable("audit_logs", {
  id: serial("id").primaryKey(),
  userId: integer("user_id").notNull(),
  action: text("action").notNull(), // "create", "update", "delete"
  table: text("table").notNull(),
  recordId: integer("record_id"),
  oldValues: text("old_values"), // JSON
  newValues: text("new_values"), // JSON
  createdAt: timestamp("created_at", { withTimezone: true }).notNull().defaultNow(),
});
```

**Middleware to log all mutations:**
```typescript
export function auditLog(action: "create" | "update" | "delete") {
  return async (req: Request, res: Response, next: NextFunction) => {
    const originalJson = res.json;
    res.json = function(data) {
      if ([201, 200].includes(res.statusCode)) {
        db.insert(auditLogsTable).values({
          userId: req.session?.userId,
          action,
          table: req.route.path,
          recordId: data?.id,
          newValues: JSON.stringify(data),
        });
      }
      return originalJson.call(this, data);
    };
    next();
  };
}
```

### 12. Dark Mode Toggle

**Add to navbar.tsx:**
```typescript
import { Moon, Sun } from "lucide-react";

export function DarkModeToggle() {
  const [isDark, setIsDark] = useState(() => 
    document.documentElement.classList.contains("dark")
  );

  const toggle = () => {
    document.documentElement.classList.toggle("dark");
    setIsDark(!isDark);
    localStorage.setItem("theme", isDark ? "light" : "dark");
  };

  return (
    <button onClick={toggle} className="p-2">
      {isDark ? <Sun /> : <Moon />}
    </button>
  );
}
```
